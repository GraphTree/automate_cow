Public Function all비우기()
Dim ws As Worksheet
    
    For Each ws In ActiveWorkbook.Worksheets
        ws.Activate
        Range("A2").Select
        If Not IsEmpty(ActiveCell.Value) Then
            Range(Selection, ActiveCell.SpecialCells(xlLastCell)).Select
            Selection.EntireRow.Delete
        End If
    Next ws
End Function

Public Function numCk()
    lastRow = Cells(Rows.Count, 1).End(xlUp).Row
    maxno = lastRow
    thisno = 2
    countno = 1
    Dim countarray()
    ReDim countarray(maxno)
    For i = thisno To maxno
        nextno = i + 1
        
        countarray(i - 2) = countno
        If (Range("C" & i).Value = Range("C" & nextno).Value) Then
            
        Else
            countno = countno + 1
        End If
        
    Next i
    
    For i = thisno To lastRow
        orderNo = Application.CountIf(Range(Cells(thisno, "C"), Cells(lastRow, "C")), Cells(i, "C"))
        Range("G" & i).Value = orderNo
    Next i
    For i = LBound(countarray) To UBound(countarray)
        For j = i + 1 To UBound(countarray)
            If (i < maxno - 1) Then
                If (countarray(i) = countarray(j)) Then
                    g1 = textcut(Range("I" & i + 2).Value)
                    g2 = textcut(Range("I" & j + 2).Value)
                    
                    If (g1 = g2) Then
                        sumno = textcutNo(Range("I" & i + 2).Value) + textcutNo(Range("I" & j + 2).Value)
                        Range("I" & i + 2).Value = Range("I" & i + 2).Value & "**(" & sumno & ") "
                        Range("I" & j + 2).Value = Range("I" & j + 2).Value & "**(" & sumno & ") "
                    End If
                    
                Else
                    Exit For
                End If
            End If
        Next j
    Next i
    
End Function

Public Function textcutNo(texts)
    Dim cut_t As Integer
    
    cutt = Split(texts, "...")
    cut_t = Left(cutt(1), InStr(cutt(1), ". .") - 1)
    textcutNo = cut_t
End Function
Public Function textcut(texts)
    
    On Error Resume Next
    cuttt = Split(texts, "(추가) ")
    If (UBound(cuttt) > 0) Then
        cutt = Left(cuttt(1), InStr(cuttt(1), "...") - 1)
    Else
        cutt = Left(texts, InStr(texts, ". .") - 1)
    End If
    On Error GoTo 0
    textcut = cutt
End Function
Public Function pay()

    Cells.Find(What:="선착불", After:=ActiveCell, LookIn:=xlFormulas, LookAt _
        :=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:= _
        False, SearchFormat:=False).Activate
    
    'Store reference of active cell to activate after selection
    Set rActive = ActiveCell
    
    'Exit if current region is a single cell
    If ActiveCell.CurrentRegion.Count = 1 Then Exit Function
    
    'Find the last used cell in the columns of the current region
    'Attempts to account for blank rows in the data by using Range.Find
    'to find the last used cell in the column of the current region.
    On Error Resume Next
        lLastRow = ActiveCell.CurrentRegion.EntireColumn.Find( _
                    What:="*", _
                    After:=ActiveCell.CurrentRegion.EntireColumn.Cells(1, 1), _
                    LookAt:=xlPart, _
                    LookIn:=xlFormulas, _
                    SearchOrder:=xlByRows, _
                    SearchDirection:=xlPrevious, _
                    MatchCase:=False).Row
    On Error GoTo 0
    
    'Find the first used row in the current region
    lFirstRow = ActiveCell.CurrentRegion.Row
    
    'Exit if any errors in setting the rows
    If lFirstRow = 0 Or lLastRow = 0 Then Exit Function
    
    'If the region contains a header row (bold font cells) then offset start by 1 row
    If Intersect(ActiveCell.CurrentRegion, ActiveCell.EntireColumn).Cells(1, 1).Font.Bold = True Then
        lFirstRow = lFirstRow + 1
    End If
    
    'Select the range within the column
    Range(Cells(lFirstRow, ActiveCell.column), Cells(lLastRow, ActiveCell.column)).Select
    
    'Activate the orginal active cell
    If Not Intersect(rActive, Selection) Is Nothing Then
        rActive.Activate
    End If
    Selection.Value = "신용"
End Function


Public Function saveThis()
NewPath = orderPath & "발주_" & Format(Now(), "MM") & "월" & Format(Now(), "dd") & "일_" & Format(Now(), "hhmmss") & ".xlsx"
ActiveWorkbook.SaveAs filename:=NewPath, FileFormat:=51
End Function


Public Function allPaste(thisSheet, homeSheet)
    Sheets(thisSheet).Select
    Range("A2").Select
    If Not IsEmpty(ActiveCell.Value) Then
        Range(Selection, ActiveCell.SpecialCells(xlLastCell)).Select
        Selection.Copy
        Sheets(homeSheet).Activate
        Range("A2").Select
        If Not IsEmpty(ActiveCell.Value) Then
            Selection.End(xlUp).Select
            Selection.End(xlDown).Select
            ActiveCell.Offset(1, 0).Select
        End If
        ActiveSheet.Paste
        Application.CutCopyMode = False
    End If
End Function

    If ActiveCell.CurrentRegion.Count = 1 Then Exit Function
    
    'Find the last used cell in the columns of the current region
    'Attempts to account for blank rows in the data by using Range.Find
    'to find the last used cell in the column of the current region.
    On Error Resume Next
        lLastRow = ActiveCell.CurrentRegion.EntireColumn.Find( _
                    What:="*", _
                    After:=ActiveCell.CurrentRegion.EntireColumn.Cells(1, 1), _
                    LookAt:=xlPart, _
                    LookIn:=xlFormulas, _
                    SearchOrder:=xlByRows, _
                    SearchDirection:=xlPrevious, _
                    MatchCase:=False).Row
    On Error GoTo 0
    
    'Find the first used row in the current region
    lFirstRow = ActiveCell.CurrentRegion.Row
    
    'Exit if any errors in setting the rows
    If lFirstRow = 0 Or lLastRow = 0 Then Exit Function
    
    'If the region contains a header row (bold font cells) then offset start by 1 row
    If Intersect(ActiveCell.CurrentRegion, ActiveCell.EntireColumn).Cells(1, 1).Font.Bold = True Then
        lFirstRow = lFirstRow + 1
    End If
    
    'Select the range within the column
    Range(Cells(lFirstRow, ActiveCell.column), Cells(lLastRow, ActiveCell.column)).Select
    
    'Activate the orginal active cell
    If Not Intersect(rActive, Selection) Is Nothing Then
        rActive.Activate
    End If
    Selection.Copy
    Sheets(homeSheet).Activate
    Cells.Find(What:=outName, After:=ActiveCell, LookIn:=xlFormulas, LookAt _
        :=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:= _
        False, SearchFormat:=False).Activate
    ActiveCell.Offset(1, 0).Select
    ActiveSheet.Paste
End Function

Public Function nameChange(newSheet, oldName, newName)
    Sheets(newSheet).Activate
    Range("A1").Select
    
    If Not Cells.Find(What:=oldName) Is Nothing Then
    Cells.Replace What:=oldName, Replacement:=newName, LookAt:=xlPart, _
        SearchOrder:=xlByRows, MatchCase:=False, SearchFormat:=False, _
        ReplaceFormat:=False
    End If
End Function


Public Function joinName(newSheet)
    Sheets(newSheet).Activate
    Range("A1").Select
    
    i = 2
    While (Range("B" & i).Value <> "")
        Range("A" & i).Value = Range("B" & i).Value & " - " & Range("C" & i).Value
        If (Range("D" & i).Value <> "") Then
            Range("A" & i).Value = Range("A" & i).Value & "... " & Range("D" & i).Value & Range("E" & i).Value & " . ."
        End If
        i = i + 1
    Wend
End Function

Function 알파벳찾기(sheetName As String, 찾을문자열 As String) As String
    Dim 대상셀 As Range
    Dim 찾은셀 As Range
    
    ' 대상 시트와 범위 설정
    Set 대상셀 = Sheets(sheetName).Rows(1)
    
    ' 대상셀에서 찾을 문자열을 포함하는 셀을 찾기
    Set 찾은셀 = 대상셀.Find(찾을문자열, LookIn:=xlValues, LookAt:=xlWhole)
    
    ' 찾은셀이 있는 경우 해당 셀의 열의 알파벳 값을 반환
    If Not 찾은셀 Is Nothing Then
        알파벳찾기 = Split(찾은셀.Address, "$")(1)
    Else
        알파벳찾기 = "해당 문자열을 찾지 못했습니다."
    End If
End Function

Sub visitSetting(직접수령여부위치, 판매채널위치)
    i = 2
    While (Range(판매채널위치 & i).Value <> "")
        If (Range(직접수령여부위치 & i).Value = "방문수령") Then
            Range(판매채널위치 & i).Value = "네이버방문수령"
        End If
        i = i + 1
    Wend
End Sub

Function gsauce()
    Dim va
    Dim vn
        lastRow = lastR()
        text1 = "  고추장맛 소스_ 40g"
        text2 = "  간장맛 소스_ 40g"
        text3 = "  뭉티기 소스_ 50g"
        
        For i = lastRow To 2 Step -1
            num = Cells(i, 9).Value
            
            va0 = Split(num, "소스 선택")
            If UBound(va0) >= 1 Then
                va = Split(va0(1), ".")
                'Cells(7, 1).Value = va(0)
                'Cells(7, 2).Value = va(1)
                'Cells(7, 3).Value = va(2)
                'Cells(7, 4).Value = va(3)
                'Cells(7, 5).Value = va(4)
                vn = va(4)
                If (va(1) = text1) Then
    
                    Rows(i).Select
                    Selection.Copy
                    Selection.Insert Shift:=xlDown
                    Cells(i, 9).Value = "z1.  고추장맛 소스_ 40g... " & vn & " . ."
                    Cells(i + 1, 9).Value = va0(0) & "... " & vn & " . ."
                ElseIf (va(1) = text2) Then
                    Rows(i).Select
                    Selection.Copy
                    Selection.Insert Shift:=xlDown
                    
                    Cells(i, 9).Value = "z2.  간장맛 소스_ 40g... " & vn & " . ."
                    Cells(i + 1, 9).Value = va0(0) & "... " & vn & " . ."
                ElseIf (va(1) = text3) Then
                    Rows(i).Select
                    Selection.Copy
                    Selection.Insert Shift:=xlDown
                    
                    Cells(i, 9).Value = "z3.  뭉티기 소스_ 50g... " & vn & " . ."
                    Cells(i + 1, 9).Value = va0(0) & "... " & vn & " . ."
                Else
                    Cells(i, 9).Value = va0(0) & "... " & va(3) & " . ."
                End If
            Else
                ' "소스 선택"이 없을 때
            End If
        Next i
End Function



Function pp()
    Cells.Find(What:="#", After:=ActiveCell, LookIn:=xlFormulas, LookAt:= _
        xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:=False _
        , SearchFormat:=False).Activate
    thisrow = ActiveCell.Row
    thisvalue = ActiveCell.Value
    thisArray = Split(thisvalue, "#")
    thislength = UBound(thisArray) - LBound(thisArray) + 1
    
    'Range("A6:A9").Value = thisArray
    'Range("A5").Value = thislength
    Rows(thisrow).Select
    For i = 1 To thislength - 1
        Selection.Copy
        Selection.Insert Shift:=xlDown
    Next i
    Application.CutCopyMode = False
    Range("I" & thisrow & ": I" & thisrow + thislength - 1).Value = Application.Transpose(thisArray)
    'Range("I" & thisrow).Select
End Function

Function setSort()
    lastRow = lastR()
    ActiveWorkbook.Worksheets("위메프변환").Sort.SortFields.Clear
    ActiveWorkbook.Worksheets("위메프변환").Sort.SortFields.Add2 Key:=Range("C2:C" & lastRow) _
        , SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    ActiveWorkbook.Worksheets("위메프변환").Sort.SortFields.Add2 Key:=Range("I2:I" & lastRow) _
        , SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    With ActiveWorkbook.Worksheets("위메프변환").Sort
        .SetRange Range("A1:O" & lastRow)
        .Header = xlYes
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin
        .Apply
    End With
End Function

Function FormatPhoneNumber(phoneNumber As String) As String
    ' 전화번호가 11자리인지 확인
    If Len(phoneNumber) = 11 Then
        ' 앞 3자리, 중간 4자리, 마지막 4자리를 추출하여 하이픈을 삽입
        FormatPhoneNumber = Mid(phoneNumber, 1, 3) & "-" & _
                            Mid(phoneNumber, 4, 4) & "-" & _
                            Mid(phoneNumber, 8, 4)
    Else
        ' 11자리가 아니면 원본을 그대로 반환
        FormatPhoneNumber = phoneNumber
    End If
End Function

Public Function BubbleSrt(StringIO As String, Ascending As Boolean)
Dim ArrayIn As Variant
Dim SrtTemp As Variant
Dim l As Long
Dim m As Long
    
    ArrayIn = Split(StringIO, "#")
    
    If Ascending = True Then
        For l = LBound(ArrayIn) To UBound(ArrayIn)
             For m = l + 1 To UBound(ArrayIn)
                 If ArrayIn(l) > ArrayIn(m) Then
                     SrtTemp = ArrayIn(m)
                     ArrayIn(m) = ArrayIn(l)
                     ArrayIn(l) = SrtTemp
                 End If
             Next m
         Next l
    Else
        For l = LBound(ArrayIn) To UBound(ArrayIn)
             For m = l + 1 To UBound(ArrayIn)
                 If ArrayIn(l) < ArrayIn(m) Then
                     SrtTemp = ArrayIn(m)
                     ArrayIn(m) = ArrayIn(l)
                     ArrayIn(l) = SrtTemp
                 End If
             Next m
         Next l
    End If
    StringIO = Join(ArrayIn, "#")
End Function

Public Function 합배송나누기(항목)
    Dim finds As String
    Dim varS
    Dim i As Long
    Cells.Find(What:=항목, After:=ActiveCell, LookIn:=xlFormulas, LookAt _
        :=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:= _
        False, MatchByte:=False, SearchFormat:=False).Activate
    ActiveCell.Offset(1, 0).Select
    For j = 0 To 2000
        If (indexOf(Selection, "#")) Then
            finds = Selection.Value
            BubbleSrt finds, True
            Selection.Value = finds
            varS = Split(finds, "#")
            For i = LBound(varS) To UBound(varS)
                Selection.Cells((i + 1), 1).Value = varS(i)
            Next i
        Else
            ActiveCell.Offset(1, 0).Select
        End If
    Next j

End Function

Function IsFileOpen(filename As String)
    Dim filenum As Integer, errnum As Integer

    On Error Resume Next   ' Turn error checking off.
    filenum = FreeFile()   ' Get a free file number.
    ' Attempt to open the file and lock it.
    Open filename For Input Lock Read As #filenum
    Close filenum          ' Close the file.
    errnum = Err           ' Save the error number that occurred.
    On Error GoTo 0        ' Turn error checking back on.

    ' Check to see which error occurred.
    Select Case errnum
        ' File is NOT already open by another user.
        Case 0
         IsFileOpen = False
        ' Error number for "Permission Denied."
        ' File is already opened by another user.
        Case 70
            IsFileOpen = True
        ' Another error occurred.
        Case Else
            Error errnum
    End Select
End Function

    ActiveSheet.Range("$A$1:$AJ$2000").RemoveDuplicates Columns:=thisc, Header:=xlYes

Public Function 전화번호맞추기()
    lastRow = lastR()
    exno1 = "010-2249-4072" '바뀐번호1
    
    For i = 2 To lastRow
        num = Cells(i, 4).Value
        leftnum = Left(num, 3)
        If (num = exno1) Then
            Cells(i, 4).Value = Cells(i, 4).Value
        End If
        If (leftnum <> "010" And leftnum <> "011") Then
            If (leftnum <> "050" And Cells(i, 3).Value <> "") Then
                Cells(i, 4).Value = Cells(i, 3).Value
            Else
                If (Cells(i, 5).Value <> "") Then
                    Cells(i, 4).Value = Cells(i, 5).Value
                End If
            End If
        End If
        
    Next i

End Function

Function indexOf(cell, textToFind)

    'cell의 값을 가져옴
    Dim val
    val = Range(cell, cell).Value



    'cell의 값 1번째 글자부터 textToFind 값 찾기
    Dim idx
    idx = InStr(1, val, textToFind, vbTextCompare)



    '인덱스 값을 리턴
    indexOf = idx

End Function
Function LastFile(Optional mask As String = "") As String
    Dim c As Integer
    Dim fname As String
    Dim fdate As Date
    Dim LastDate As Date
    Dim NotAttr As Integer
    
    NotAttr = vbSystem + vbDirectory
        ' 시스템 파일이나 디렉토리명은 처리 대상에서 제외하기 위함
    For c = Len(mask) To 1 Step -1
        If Mid(mask, c, 1) = "\" Then Exit For
    Next c
    If c > 0 Then
        ChDir Left(mask, c - 1)
        mask = Mid(mask, c + 1)
    End If
    LastDate = 1
    fname = Dir(mask)   ' 지정한 폴더에서 첫번째 파일 찾음
    Do While fname <> ""   ' Start the loop.
        If Not (GetAttr(fname) And NotAttr) Then
            ' 시스템파일이나 디렉토리명은 제외하고 처리
            fdate = FileDateTime(fname)
            ' 해당 파일의 생성 날짜/시각을 읽어서 최댓값을 검색함
            If fdate > LastDate Then
                LastDate = fdate
                LastFile = fname
            End If
        End If
        fname = Dir()   ' 다음 대상 파일명 찾기
     Loop
End Function
